# 自定义设置
/粘贴时格式化/
"editor.formatOnPaste": true
/改变字体大小/
"editor.fontSize": 18
/设置缩放等级/
"window.zoomLevel": 5
/编程字体连字/
"editor.fontFamily": "Fira Code",
"editor.fontLigatures": true
/自动保存/： 你还可以使用最高级别的自动保存，在File > Auto Save设置
"files.autoSave": "afterDelay"
/保存时格式化/
"editor.formatOnSave": true,
/修改tab长度/
"editor.tabSize": 4
/空白符号显示/
"editor.renderWhitespace": "all"
